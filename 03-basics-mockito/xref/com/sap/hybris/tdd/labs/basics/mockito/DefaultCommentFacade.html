1   package com.sap.hybris.tdd.labs.basics.mockito;

3   import java.util.List;
4   import java.util.stream.Collectors;
5   
6   import com.sap.hybris.tdd.labs.basics.mockito.model.CommentModel;
7   import com.sap.hybris.tdd.labs.basics.mockito.model.Model;
8   import com.sap.hybris.tdd.labs.basics.mockito.model.UserModel;
9   import com.sap.hybris.tdd.labs.basics.mockito.service.CommentService;
10  import com.sap.hybris.tdd.labs.basics.mockito.service.PermissionService;
11  import com.sap.hybris.tdd.labs.basics.mockito.service.UserService;
12  
13  public class DefaultCommentFacade implements CommentFacade {
14  
15      private final UserService userService;
16      private final PermissionService permissionService;
17      private final CommentService commentService;
18  
19      public DefaultCommentFacade(final UserService userService, final PermissionService permissionService,
20              final CommentService commentService) {
21          this.userService = userService;
22          this.permissionService = permissionService;
23          this.commentService = commentService;
24      }
25  
26      @Override
27      public boolean addComment(final Model model, final String commentMessage) {
28          final UserModel currentUser = userService.getCurrentUser();
29          if (permissionService.canUserCreate(currentUser, CommentModel.class)) {
30              final CommentModel comment = commentService.createComment(commentMessage);
31              commentService.addComment(model, comment);
32              return true;
33          }
34          return false;
35      }
36  
37      @Override
38      public List<CommentModel> getComments(final Model model) {
39          final UserModel currentUser = userService.getCurrentUser();
40          final List<CommentModel> comments = commentService.getComments(model);
41          return comments.stream()//
42                  .filter(comment -> permissionService.canUserRead(currentUser, comment))//
43                  .collect(Collectors.toList());
44      }
45  }
