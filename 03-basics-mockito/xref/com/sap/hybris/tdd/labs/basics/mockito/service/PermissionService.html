1   package com.sap.hybris.tdd.labs.basics.mockito.service;
2   
3   import com.sap.hybris.tdd.labs.basics.mockito.model.Model;
4   import com.sap.hybris.tdd.labs.basics.mockito.model.UserModel;
5   
6   /**
7    * Provides functionality to verify if users have rights to read/create/modify other models.
8    */
9   public interface PermissionService {
10  
11      /**
12       * Tests whether a user has granted permissions to read a model.
13       * @param user the user which wants to read the model.
14       * @param model the model which the user wants to read.
15       * @return {@code true} whether the user can read the model, otherwise {@code false}.
16       */
17      boolean canUserRead(UserModel user, Model model);
18  
19      /**
20       * Tests whether a user has granted permissions to create a new model.
21       * @param user the user which wants to create the new model.
22       * @param modelClass the model class of which the user wants to create a new instance.
23       * @return {@code true} whether the user can create the new model, otherwise {@code false}.
24       */
25      boolean canUserCreate(UserModel user, Class<? extends Model> modelClass);
26  }
